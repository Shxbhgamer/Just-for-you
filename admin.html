const adminForm = document.getElementById("adminForm"); adminForm.addEventListener("submit", async (e) => { e.preventDefault(); const loginData = { email: adminForm.email.value, password: adminForm.password.value, }; try { const res = await fetch("http://127.0.0.1:5000/api/admin/login", { method: "POST", headers: { "Content-Type": "application/json" }, body: JSON.stringify(loginData), }); const data = await res.json(); if (res.ok) { localStorage.setItem("adminToken", data.token); alert("Login successful!"); window.location.href = "adminDashboard.html"; } else { alert(data.msg || "Login failed"); } } catch (err) { console.error(err); alert("Error logging in."); } });
